)

Create Table CardRarities
(
	Id int Identity(1,1) primary key,
	Rarity nvarchar(50) NOT NULL
)
go

insert into CardRarities (Rarity)
values
('Common'),
('Uncommon'),
('Rare'),
('Unique'),
('Legendary')
go

Create Table Cards
(
	Id int Identity(1,1) primary key,
	Name nvarchar(50) NOT NULL,
	CardTypeId int foreign key references CardTypes(Id),
	ResourceRequirement int NULL,
	CardText nvarchar(200) NULL,
	Image image NULL,
	AttackValue int NULL,
	StrengthValue int NULL,
	CardRarityId int foreign key references CardRarities(Id)
)
go

Create Table CardsAbilitiesValues
(
	Id int Identity(1,1) primary key,
	CardId int foreign key references Cards(Id) NOT NULL,
	CardAbilityTypeId int foreign key references CardAbilityTypes(Id) NULL,
	AbilityValue int NULL
)
go
